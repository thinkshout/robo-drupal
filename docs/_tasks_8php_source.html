247

249  $this->taskRsync()
250  ->fromPath("./")
251  ->toPath("$tmpDir/deploy")
252  ->args('-a', '-v', '-z', '--no-group', '--no-owner')
253  ->excludeVcs()

256  ->exclude('sites/default/files')

258  ->run();

261  $this->taskRsync()
262  ->fromPath("$tmpDir/$hostDirName/.git")
263  ->toPath("$tmpDir/deploy")
264  ->args('-a', '-v', '-z', '--no-group', '--no-owner')

273  ->preferDist()
274  ->run();

497  public function pantheonInstall() {
498  $admin_name = $this->projectProperties['admin_name'];
499 
500  $terminus_site_env = $this->getPantheonSiteEnv();
501  // Pantheon wants the site in SFTP for installs.
502  $this->_exec("terminus connection:set $terminus_site_env sftp");
503 
504  // Even in SFTP mode, the settings.php file might have too restrictive
505  // permissions. We use SFTP to chmod the settings file before installing.
506  $sftp_command = trim(exec("terminus connection:info --field=sftp_command $terminus_site_env"));
507  $sftp_command = str_replace('sftp', 'sftp -b -', $sftp_command);
508  // Use webroot to find settings.php assume webroot is the gitroot if no
509  // webroot is specified.
510  if (getenv('TS_WEB_ROOT')) {
511  $web_root = getenv('TS_WEB_ROOT') . '/';
512  }
513  else {
514  $web_root = '';
515  }
516  $default_dir = 'code/' . $web_root . 'sites/default';
517  // We use 755 instead of 644 so settings.php is executable, and the
518  // directory is stat-able (otherwise we can't chmod the php file)
519  $sftp_command .= ' << EOF
520 chmod 755 ' . $default_dir . '
521 chmod 755 ' . $default_dir . '/settings.php';
522  // Note that we don't use $this->_exec on purpose. SFTP command fails
523  // with that operation: a fix would be great but this actually works.
524  exec($sftp_command);
525 
526  // Run the installation.
527  $result = $this->trueFreshInstall("terminus remote:drush $terminus_site_env --");
528  // Put the site back into git mode.
529  $this->_exec("terminus connection:set $terminus_site_env git");
530 
531  if ($result->wasSuccessful()) {
532  $this->say('Install complete');
533  }
534 
535  return $result;
536  }
